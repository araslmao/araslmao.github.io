// ==UserScript==
// @name         Roblox Connect to Friends
// @namespace    http://tampermonkey.net/
// @version      1.1
// @description  Changes "Connect", "Connection", and "Connections" to "Friends" on Roblox.com
// @author       araslmao on YouTube
// @match        https://*.roblox.com/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    // Function to replace text in the DOM
    function replaceText(node) {
        if (node.nodeType === Node.TEXT_NODE) {
            node.textContent = node.textContent
                .replace(/\bConnect\b/gi, 'Friends')
                .replace(/\bConnection\b/gi, 'Friends')
                .replace(/\bConnections\b/gi, 'Friends');
        } else {
            node.childNodes.forEach(replaceText);
        }
    }

    // Initial replacement
    replaceText(document.body);

    // Observe DOM changes for dynamic content
    const observer = new MutationObserver((mutations) => {
        mutations.forEach((mutation) => {
            mutation.addedNodes.forEach(replaceText);
        });
    });

    // Start observing
    observer.observe(document.body, {
        childList: true,
        subtree: true
    });
})();
